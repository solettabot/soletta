158 {
159  const char *p = (const char *)mem;
160 
161  if (strchr("{}[],:tfn\"", *p))

218 static inline bool
219 _sol_json_loop_helper_generic(struct sol_json_scanner *scanner, struct sol_json_token *token,
220  enum sol_json_type end_type, enum sol_json_loop_reason *reason)
221 {
222  if (*reason != SOL_JSON_LOOP_REASON_OK)
223  return false;
224 
225  if (!sol_json_scanner_next(scanner, token)) {
226  *reason = SOL_JSON_LOOP_REASON_INVALID;
227  return false;
228  }
229 
230  if (sol_json_token_get_type(token) == end_type) {
231  *reason = SOL_JSON_LOOP_REASON_OK;
232  return false;
233  }
234 
235  if (sol_json_token_get_type(token) == SOL_JSON_TYPE_ELEMENT_SEP) {
236  if (!sol_json_scanner_next(scanner, token)) {
237  *reason = SOL_JSON_LOOP_REASON_INVALID;
238  return false;
239  }
240  }
241 
242  return true;
243 }
